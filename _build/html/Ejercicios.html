from sympy import symbols, Function, dsolve

# Smbolos
t, g, k , m, v0= symbols("t, g, k, m, v0")

# v es funcin del tiempo
v = Function('v')(t)

# Derivada de v
dvdt = v.diff(t)

# Ecuacin diferencial
eq = dvdt - g + k*v/m

# Solucin de la EDO con la condicin inicial v(0)=v0
dsolve(eq, v, ics={v.subs(t, 0): v0})

import numpy as np
import matplotlib.pyplot as plt
from scipy.integrate import solve_ivp

# Parmetros de la ley de enfriamiento de Newton
k = -0.1  # Constante de enfriamiento

# Funcin que define la temperatura del medio variable en el tiempo
def T_m(t):
    return 20 + 10 * np.sin(0.1 * t)

# Definicin de la EDO para la ley de enfriamiento de Newton
def Newton(t, T):
    return k * (T - T_m(t))

# Condicin inicial
T0 = 100  # Temperatura inicial del objeto

# Intervalo de tiempo para la simulacin
t_span = (0, 100)
t_eval = np.linspace(t_span[0], t_span[1], 1000)

# Resolver la EDO
sol = solve_ivp(Newton, t_span, [T0], t_eval=t_eval)

# Graficar los resultados
plt.plot(sol.t, sol.y[0], label='Temperatura del objeto')
plt.plot(sol.t, T_m(sol.t), label='Temperatura del medio', linestyle='dashed')
plt.xlabel('Tiempo')
plt.ylabel('Temperatura')
plt.legend()
plt.title('Ley de Enfriamiento de Newton con Temperatura del Medio Variable')
plt.show()

def f(x, y):
    return np.sin(x)*np.cos(y)

def plot_slope_field(a=-5, b=5, c=-5, d=5, n=20, x0=0, y0=1):
    x = np.linspace(a, b, n)
    y = np.linspace(c, d, n)
    X, Y = np.meshgrid(x, y)
    U = 1
    V = f(X, Y)
    
    # Normalizacin de las flechas
    N = np.sqrt(U**2 + V**2)
    U2, V2 = U/N, V/N
    
    plt.figure(figsize=(8, 6))
    plt.quiver(X, Y, U2, V2, angles="xy")
    plt.xlabel('x')
    plt.ylabel('y')
    plt.title('Campo de pendientes')
    plt.xlim([a, b])
    plt.ylim([c, d])
    plt.grid()
    
    # Resolucin de la EDO con condicin inicial en ambas direcciones.
    def ode_system(t, y):
        return f(t, y)
    
    # Hacia adelante en el tiempo
    t_span_forward = [x0, b]
    t_eval_forward = np.linspace(x0, b, 100)
    sol_forward = solve_ivp(ode_system, t_span_forward, [y0], t_eval=t_eval_forward)
    
    # Hacia atrs en el tiempo
    t_span_backward = [x0, a]
    t_eval_backward = np.linspace(x0, a, 100)
    sol_backward = solve_ivp(ode_system, t_span_backward, [y0], t_eval=t_eval_backward)
    
    # Grfico de la solucin
    plt.plot(sol_forward.t, sol_forward.y[0], 'r', label=f'Solucin con CI: y({round(x0,2)})={round(y0,2)}')
    plt.plot(sol_backward.t, sol_backward.y[0], 'r')
    plt.scatter([x0], [y0], color='red')  # Condicin inicial
    plt.legend()
    plt.show()

import numpy as np
import matplotlib.pyplot as plt
from scipy.integrate import solve_ivp

# Definimos la funcin f(x, y)
def f(x, y):
    return 2 * np.sqrt(y)

# Condiciones iniciales
x0 = 2
y0 = 0

# Intervalo de integracin
x_span = [x0, 5]

# Resolver la EDO usando solve_ivp
sol = solve_ivp(f, x_span, [y0], method='RK45', t_eval=np.linspace(x0, 5, 100))

# Solucin analtica: y(x) = (x - 2)^2
x = np.linspace(2, 5, 100)
y_analitica = (x - 2)**2

# Solucin trivial: y(x) = 0
y_trivial = np.zeros_like(x)

# Visualizar las soluciones
plt.plot(sol.t, sol.y[0], label='Solucin numrica (solve_ivp)')
plt.plot(x, y_analitica, 'r--', label='Solucin analtica: $(x-2)^2$')
plt.plot(x, y_trivial, 'g--', label='Solucin trivial: $y(x) = 0$')
plt.xlabel('x')
plt.ylabel('y')
plt.title(r'Soluciones para $y\' = 2\sqrt{y}$ con $y(2) = 0$')
plt.legend()
plt.grid(True)
plt.show()

import numpy as np
import matplotlib.pyplot as plt
from scipy.integrate import solve_ivp

# Parmetros del sistema
m = 1.0   # masa
k = 1.0   # constante del resorte
c = 0.5   # coeficiente de amortiguamiento
F0 = 1  # amplitud de la fuerza externa (puede ser 0 si no hay fuerza externa)
omega = 1.0  # frecuencia de la fuerza externa

# Definir el sistema de ecuaciones diferenciales: Una EDO de 2do orden se "separa" como un sistema para aplicar el
#el comando solve_ivp
def masa_resorte(t, y, m, k, beta, F0, gamma):
    x, v = y
    dxdt = v
    dvdt = -(k/m) * x - (beta/m) * v + (F0/m) * np.cos(gamma * t) #Fuerza externa F0*cos(gamma*t): se puede cambiar
    return [dxdt, dvdt]

# Condiciones iniciales
y0 = [1.0, 0.0]  # posicin inicial x(0)=1, velocidad inicial x'(0)=0

# Tiempo de simulacin
t_span = (0, 30)
t_eval = np.linspace(*t_span, 500)

# Resolver las ecuaciones diferenciales para los tres casos
sol_libre_sin_amort = solve_ivp(masa_resorte, t_span, y0, args=(m, k, 0, 0, 0), t_eval=t_eval)
sol_libre_amort = solve_ivp(masa_resorte, t_span, y0, args=(m, k, c, 0, 0), t_eval=t_eval)
sol_forzado = solve_ivp(masa_resorte, t_span, y0, args=(m, k, c, F0, omega), t_eval=t_eval)

# Graficar los resultados en un mismo grfico
plt.figure(figsize=(10, 6))
plt.plot(sol_libre_sin_amort.t, sol_libre_sin_amort.y[0], label='Movimiento libre sin amortiguacin', color='green')
plt.plot(sol_libre_amort.t, sol_libre_amort.y[0], label='Movimiento libre amortiguado', color='blue')
plt.plot(sol_forzado.t, sol_forzado.y[0], label='Movimiento forzado', linestyle='dashed', color='red')

plt.title('Comparacin de los Movimientos: Libre, Amortiguado y Forzado')
plt.xlabel('Tiempo (t)')
plt.ylabel('Posicin (x)')
plt.legend()
plt.grid(True)
plt.tight_layout()
plt.show()
